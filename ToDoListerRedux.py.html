1    import tkinter as tk

5    class ToDoLister(tk.Tk):

7            super().__init__()

10           self.title("To Do Lister")
11   
12           self.create_widgets()
13   
14       def create_widgets(self):
15           self.task_input = tk.Entry(self, width=50)
16           self.task_input.pack(pady=10)
17   
18           self.add_task_button = tk.Button(self, text="Add task", command=self.add_task)
19           self.add_task_button.pack(pady=10)
20   
21           self.tasks_listbox = tk.Listbox(self, selectmode=tk.SINGLE)
22           self.tasks_listbox.pack(pady=10)
23   
24           self.button_frame = tk.Frame(self)
25           self.button_frame.pack(pady=10)
26   
27           self.edit_task_button = tk.Button(self.button_frame, text="Edit", command=self.edit_task)
28           self.edit_task_button.grid(row=0, column=0, padx=10, pady=10)
29   
30           self.delete_task_button = tk.Button(self.button_frame, text="Delete Task", command=self.delete_task)
31           self.delete_task_button.grid(row=0, column=1, padx=10, pady=10)
32   
33           self.save_button = tk.Button(self.button_frame, text="Save",command=self.save_tasks)
34           self.save_button.grid(row=1, column=0, padx=10, pady=10)
35   
36           self.load_button = tk.Button(self.button_frame, text="Load", command=self.load_tasks)
37           self.load_button.grid(row=1, column=1, padx=10, pady=10)
38   
39       def add_task(self):
40           task = self.task_input.get()
41           if task:
42               self.tasks_listbox.insert(tk.END, task)
43               self.task_input.delete(0, tk.END)
44   
45       def edit_task(self):
46           task_index = self.tasks_listbox.curselection()
47           if task_index:
48               new_task = self.task_input.get()
49               if new_task:
50                   self.tasks_listbox.delete(task_index)
51                   self.tasks_listbox.insert(task_index, new_task)
52                   self.task_input.delete(0, tk.END)
53   
54       def delete_task(self):
55               task_index = self.tasks_listbox.curselection()
56               if task_index:
57                   self.tasks_listbox.delete(task_index)
58   
59       def save_tasks(self):
60           tasks = self.tasks_listbox.get(0, tk.END)
61           file_path = filedialog.asksaveasfilename(defaultextension=".txt")
62           if file_path:
63               with open(file_path, "w") as file:
64                   for task in tasks:
65                       file.write(task + "\n")
66   
67       def load_tasks(self):
68           file_path = filedialog.askopenfilename(filetypes=[("Text Files", "*.txt")])
69   
70           if file_path:
71               with open(file_path, "r") as file:
72                   tasks = [line.strip() for line in file.readlines()]
73   
74               self.tasks_listbox.delete(0, tk.END)
75   
76               for task in tasks:
77                   self.tasks_listbox.insert(tk.END, task)
